CREATE TABLE MEMBERSHIP 
(
	MEM_NUM INT PRIMARY KEY, 
	MEM_FNAME VARCHAR(50),
    MEM_LNAME VARCHAR(50),
    MEM_STREET VARCHAR(50),
    MEM_CITY VARCHAR(50),
    MEM_STATE VARCHAR(50),
    MEM_ZIP CHAR(3),
    MEM_BALANCE DECIMAL(19,4)
);

CREATE TABLE RENTAL
(
	RENT_NUM INT PRIMARY KEY,
    RENT_DATE DATE,
    MEM_NUM INT   
);

CREATE TABLE PRICE
(
	PRICE_CODE INT PRIMARY KEY,
    PRICE_DESCRIPTION VARCHAR(50),
    PRICE_RENTFEE DECIMAL(19,4),
    PRICE_DAILYLATEFEE DECIMAL(19,4)
);

CREATE TABLE MOVIE
(
	MOVIE_NUM INT PRIMARY KEY,
    MOVIE_TITLE VARCHAR(50),
    MOVIE_YEAR VARCHAR(4),
    MOVIE_COST DECIMAL(19,4),
    MOVIE_GENRE VARCHAR(50),
    PRICE_CODE INT,
    FOREIGN KEY (PRICE_CODE) REFERENCES PRICE(PRICE_CODE)
);

CREATE TABLE VIDEO
(
	VID_NUM INT PRIMARY KEY,
    VID_INDATE DATE,
    MOVIE_NUM INT,
    FOREIGN KEY (MOVIE_NUM) REFERENCES MOVIE(MOVIE_NUM)
);

CREATE TABLE DETAILRENTAL
(
	RENT_NUM INT PRIMARY KEY,
    VID_NUM INT,
    DETAIL_FEE DECIMAL(19,4),
    DETAIL_DUEDATE DATE,
    DETAIL_RETURNDATE DATE,
    DETAIL_DAILYLATEFEE DECIMAL(19,4)
);
